Figure: US map with states filled by presence/absence of isolates

Florian Lecorvaisier

2023-01-06

library(extrafont)
library(ggplot2)
library(rnaturalearth)
library(sf)
loadfonts(quiet = T)

data_prn <- read.table("data/data-lefrancq-2022.txt", header = T, sep = "\t")

data_prn <- data_prn[data_prn$country == "US", ]

data_prn <- na.omit(data_prn[, c("year_isolated", "region", "prn_def")])

data_prn$region[data_prn$region == "New York State"] <- "New York"
data_prn$region[data_prn$region == "Washington State"] <- "Washington"

states_names <- read.table("data/states-names.txt", h = T, sep = "\t")
for (name in states_names$long) {
  data_prn$region[data_prn$region == name] <- states_names$short[states_names$long == name]
}

us_map <- ne_states(country = "united states of america", returnclass = "sf")
us_map <- us_map[!(us_map$postal %in% c("AK", "HI")), ]

for (s in unique(us_map$postal)) {
  if (s %in% data_prn$region) {
    if (max(data_prn$year_isolated[data_prn$region == s]) < 
        min(data_prn$year_isolated[data_prn$prn_def == "-"])) {
      us_map$has_data[us_map$postal == s] <- "bPRNm"
    }
    else us_map$has_data[us_map$postal == s] <- "aPRNm"
  }
  else us_map$has_data[us_map$postal == s] <- "noData"
}

ggplot(us_map) +
  geom_sf(aes(fill = has_data, linetype = has_data), color = "grey50") +
  coord_sf(crs = "+proj=lcc +lon_0=-100 +lat_1=28 +lat_2=40") +
  scale_fill_manual(values = c("#3D405B", "#636785", "#F4F1DE"),
                    labels = c(paste("At least one isolate since", min(data_prn$year_isolated[data_prn$prn_def == "-"])), 
                               paste("Isolate(s) only before", min(data_prn$year_isolated[data_prn$prn_def == "-"])), 
                               "No isolate at all")) +
  labs(fill = element_blank()) +
  guides(linetype = "none", fill = guide_legend(byrow = T)) +
  theme_void() +
  theme(text = element_text(family = "Ubuntu"),
        legend.position = c(0.2, 0.14),
        legend.spacing.y = unit(2, "mm"))
